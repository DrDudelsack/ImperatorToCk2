	# Roman character may convert to Italian
	irck2_convert_to_italian = {
		only_playable = yes

		filter = self
		ai_target_filter = self
		ai_check_interval = 12

		potential = {
			is_playable = yes
			culture = roman
			capital_scope = {
				region = custom_irck2_italy
			}
			any_realm_province = {
				culture = italian
			}
			OR = {
				year = 500
				top_liege = {
					NOT = { culture = roman }
				}
			}
			NOT = { government = chinese_imperial_government }
			NOT = {
				has_alternate_start_parameter = { key = culture_names value = random }
			}
		}
		allow = {
			culture = roman
			capital_scope = {
				region = custom_irck2_italy
			}
			any_realm_province = {
				culture = italian
			}
		}
		effect = {
			any_courtier = {
				limit = {
					culture = ROOT
					employer = { character = ROOT }
					dynasty = ROOT
				}
				culture = italian
			}
			culture = italian
		}
		revoke_allowed = {
			always = no
		}
		ai_will_do = {
			factor = 1

			modifier = { #If Ruler has 2 king titles, is emperor, or is under Roman ruler, will not do
				factor = 0
				OR = {
					top_liege = {
						culture = roman
						OR = {
							tier = emperor
							#num_of_king_titles = 2
						}
					}

					AND = {
						independent = no
						top_liege = { culture = roman }
					}
				}
			}

			modifier = {
				#factor = 0.5 # Slow it down a bit
			}

			modifier = {
				factor = 0.01
				capital_scope = {
					NOT = { culture = italian }
				}
			}
			modifier = {
				factor = 2.0
				liege = { culture = italian }
			}
			modifier = {
				factor = 1.1
				num_culture_realm_provs = {
					value = 2
					culture = italian
				}
			}
			modifier = {
				factor = 1.1
				num_culture_realm_provs = {
					value = 4
					culture = italian
				}
			}
			modifier = {
				factor = 1.1
				num_culture_realm_provs = {
					value = 6
					culture = italian
				}
			}
			modifier = {
				factor = 1.1
				num_culture_realm_provs = {
					value = 8
					culture = italian
				}
			}
			modifier = {
				factor = 1.1
				num_culture_realm_provs = {
					value = 10
					culture = italian
				}
			}
		}
	}
