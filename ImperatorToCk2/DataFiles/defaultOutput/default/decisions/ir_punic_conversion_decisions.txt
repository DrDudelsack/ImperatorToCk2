	# Punic character may convert to Taliaite
	irck2_convert_to_taliaite = {
		only_playable = yes

		filter = self
		ai_target_filter = self
		ai_check_interval = 12

		potential = {
			is_playable = yes
			culture = carthaginian
			capital_scope = {
				region = custom_irck2_italy
			}
			any_realm_province = {
				culture = taliaite
			}
			OR = {
				year = 500
				top_liege = {
					NOT = { culture = carthaginian }
				}
			}
			NOT = { government = chinese_imperial_government }
			NOT = {
				has_alternate_start_parameter = { key = culture_names value = random }
			}
		}
		allow = {
			culture = carthaginian
			capital_scope = {
				region = custom_irck2_italy
			}
			any_realm_province = {
				culture = taliaite
			}
		}
		effect = {
			any_courtier = {
				limit = {
					culture = ROOT
					employer = { character = ROOT }
					dynasty = ROOT
				}
				culture = taliaite
			}
			culture = taliaite
		}
		revoke_allowed = {
			always = no
		}
		ai_will_do = {
			factor = 1

			modifier = { #If Ruler has 2 king titles, is emperor, or is under Punic ruler, will not do
				factor = 0
				OR = {
					top_liege = {
						culture = carthaginian
						OR = {
							tier = emperor
							#num_of_king_titles = 2
						}
					}

					AND = {
						independent = no
						top_liege = { culture = carthaginian }
					}
				}
			}

			modifier = {
				#factor = 0.5 # Slow it down a bit
			}

			modifier = {
				factor = 0.01
				capital_scope = {
					NOT = { culture = taliaite }
				}
			}
			modifier = {
				factor = 2.0
				liege = { culture = taliaite }
			}
			modifier = {
				factor = 1.1
				num_culture_realm_provs = {
					value = 2
					culture = taliaite
				}
			}
			modifier = {
				factor = 1.1
				num_culture_realm_provs = {
					value = 4
					culture = taliaite

			}
			modifier = {
